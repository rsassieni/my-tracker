        <div class="text-5xl font-extrabold text-blue-600 mb-6" id="countDisplay">0</div>

        

        <button onclick="incrementStreak()" 

            class="w-full bg-blue-600 hover:bg-blue-700 text-white font-bold py-4 px-4 rounded-lg transition duration-200">

            Finished & Focused

        </button>


        <button onclick="resetStreak()" class="mt-8 text-xs text-gray-400 underline">

            Reset Counter

        </button>

    </div>


    <script>

        // Load the count when the page opens

        let count = localStorage.getItem('daveningStreak') || 0;

        document.getElementById('countDisplay').innerText = count;


        function incrementStreak() {

            count++;

            document.getElementById('countDisplay').innerText = count;

            localStorage.setItem('daveningStreak', count);

        }


        function resetStreak() {

            if(confirm("Are you sure you want to clear your progress?")) {

                count = 0;

                document.getElementById('countDisplay').innerText = count;

                localStorage.setItem('daveningStreak', 0);

            }

        }

    </script>
